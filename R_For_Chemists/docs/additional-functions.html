## Now let's look at a few new functions:

# mutate function ------------------------------
# Add a column "kilometers per liter (klp)" and calculate using mpg
EUROPE_mtcars <- mutate(mtcars,kpl = mpg*0.4251)

# %in% operator --------------------------------
# Find cars in the horsepower mid-range
mid_strength <- filter(mtcars,hp %in% 100:200)

# case_when() function -------------------------
# Classify the fuel efficiency of 
efficiency <- mutate(mtcars,fuel_efficient = case_when(
                     mpg <= 15 ~ "Bad", mpg > 15 & mpg < 25 ~ "ok", mpg >= 25 ~ "Good"))

# weighted.mean() ------------------------------
# In cases when the a variable of weights are provided.
# Common with survey data when a subset of the population
# is used to estimate the whole population. A weight is
# applied to "scale up" the survey size.

# population_estimate <- summarize(data, utility = weighted.mean(utility,weight))
population_estimate <- summarize(data, COSTELEC = weighted.mean(COSTELEC,HHWT))

# Review of filter, group_by, summarize, and %>%
# How many cars with 8 cylinders have automatic transmissions?
mtcars %>%
  filter(cyl > 6) %>%
  group_by(am) %>%
  summarize(n = n())
